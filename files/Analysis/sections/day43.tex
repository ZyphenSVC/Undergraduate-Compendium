\begin{proposition}{
    \[
        \abs{\int_a^b f(x)\,dx}\leq\int_a^b\abs{f(x)}\,dx.
    \]
}
    For all $x\in[a,b]$,
    \[
        -\abs{f(x)} \leq f(x) \leq \abs{f(x)}.
    \]

    By monotonicity of the integral,
    \[
        -\int_a^b \abs{f(x)}\,dx \leq \int_a^b f(x)\,dx \leq \int_a^b \abs{f(x)}\,dx.
    \]

    Hence
    \[
        \abs{\int_a^b f(x)\,dx}\leq\int_a^b\abs{f(x)}\,dx.
    \]
\end{proposition}
